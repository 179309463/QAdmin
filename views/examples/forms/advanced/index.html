Font Icon Picker（图标选择器） 官网

默认

添加 data-plugin="iconpicker" 即可调用图标选择器

在输入组中使用

添加 data-placement="bottomRight"

在搜索框中使用

添加 data-input-search="true"

下拉模式

添加 data-toggle="dropdown"

内联模式

添加 data-placement="inline"

自定义配置

{
    //...
    title: '自定义配置',
    icons: ['fa fa-github', 'fa fa-heart', 'fa fa-html5', 'fa fa-css3'],
    selectedCustomClass: 'badge badge-success',
    mustAccept:true,
    placement:'bottomRight',
    showFooter:true,
    //...
}

自定义配置

{
    //...
    title: '使用 glypghicons',
    icons: $.merge([
        'glyphicon-home',
        'glyphicon-repeat',
        'glyphicon-search',
        'glyphicon-arrow-left',
        'glyphicon-arrow-right',
        'glyphicon-star'],
        $.iconpicker.defaultOptions.icons),
    fullClassFormatter: function(val){
        if(val.match(/^fa fa-/)){
            return 'fa '+val;
        }else{
            return 'glyphicon '+val;
        }
    }
    //...
}
